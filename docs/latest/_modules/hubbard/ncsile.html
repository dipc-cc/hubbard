[docs]    def write_density(self, infolabel, group, n):
        # Create group
        g = self._crt_grp(self, group)
        g.info = infolabel

        # Create dimensions
        self._crt_dim(self, 'ncomp', n.shape[0])
        self._crt_dim(self, 'norb', n.shape[1])

        # Write variable n
        v = self._crt_var(g, 'n', ('f8', 'f8'), ('ncomp', 'norb'))
        v.info = 'Densities'
        g.variables['n'][:] = n
